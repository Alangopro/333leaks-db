-- Toggle state
local vehicleFlyEnabled = false
local currentVehicle = nil

-- Function to initiate vehicle fly
local function startVehicleFly(vehicle)
    -- Ensure LocalPlayer and mouse are available
    repeat wait() until game.Players.LocalPlayer and game.Players.LocalPlayer.Character and game.Players.LocalPlayer:GetMouse()
    
    local Players = game:GetService("Players")
    local IYMouse = game.Players.LocalPlayer:GetMouse()
    local getRoot = function(object) local R = object:FindFirstChild("HumanoidRootPart") if R then return R else return object.Head end end

    local FLYING = false
    local QEfly = true
    local iyflyspeed = 1
    local vehicleflyspeed = 1
    local T = getRoot(vehicle)
    local CONTROL = {F = 0, B = 0, L = 0, R = 0, Q = 0, E = 0}
    local lCONTROL = {F = 0, B = 0, L = 0, R = 0, Q = 0, E = 0}
    local SPEED = 0

    local BG = Instance.new('BodyGyro')
    local BV = Instance.new('BodyVelocity')

    BG.P = 9e4
    BG.Parent = T
    BV.Parent = T
    BG.maxTorque = Vector3.new(9e9, 9e9, 9e9)
    BG.cframe = T.CFrame
    BV.velocity = Vector3.new(0, 0, 0)
    BV.maxForce = Vector3.new(9e9, 9e9, 9e9)

    -- Functions for handling key events
    local flyKeyDown = IYMouse.KeyDown:Connect(function(KEY)
        if KEY:lower() == 'w' then
            CONTROL.F = (vehicleFlyEnabled and vehicleflyspeed or iyflyspeed)
        elseif KEY:lower() == 's' then
            CONTROL.B = -(vehicleFlyEnabled and vehicleflyspeed or iyflyspeed)
        elseif KEY:lower() == 'a' then
            CONTROL.L = -(vehicleFlyEnabled and vehicleflyspeed or iyflyspeed)
        elseif KEY:lower() == 'd' then
            CONTROL.R = (vehicleFlyEnabled and vehicleflyspeed or iyflyspeed)
        elseif QEfly and KEY:lower() == 'e' then
            CONTROL.Q = (vehicleFlyEnabled and vehicleflyspeed or iyflyspeed) * 2
        elseif QEfly and KEY:lower() == 'q' then
            CONTROL.E = -(vehicleFlyEnabled and vehicleflyspeed or iyflyspeed) * 2
        end
    end)

    local flyKeyUp = IYMouse.KeyUp:Connect(function(KEY)
        if KEY:lower() == 'w' then
            CONTROL.F = 0
        elseif KEY:lower() == 's' then
            CONTROL.B = 0
        elseif KEY:lower() == 'a' then
            CONTROL.L = 0
        elseif KEY:lower() == 'd' then
            CONTROL.R = 0
        elseif KEY:lower() == 'e' then
            CONTROL.Q = 0
        elseif KEY:lower() == 'q' then
            CONTROL.E = 0
        end
    end)

    -- Main function to manage flying
    local function startFlying()
        FLYING = true
        task.spawn(function()
            repeat wait()
                if not vehicleFlyEnabled and Players.LocalPlayer.Character:FindFirstChildOfClass('Humanoid') then
                    Players.LocalPlayer.Character:FindFirstChildOfClass('Humanoid').PlatformStand = true
                end
                if CONTROL.L + CONTROL.R ~= 0 or CONTROL.F + CONTROL.B ~= 0 or CONTROL.Q + CONTROL.E ~= 0 then
                    SPEED = 50
                elseif not (CONTROL.L + CONTROL.R ~= 0 or CONTROL.F + CONTROL.B ~= 0 or CONTROL.Q + CONTROL.E ~= 0) and SPEED ~= 0 then
                    SPEED = 0
                end
                if (CONTROL.L + CONTROL.R) ~= 0 or (CONTROL.F + CONTROL.B) ~= 0 or (CONTROL.Q + CONTROL.E) ~= 0 then
                    BV.velocity = ((workspace.CurrentCamera.CoordinateFrame.lookVector * (CONTROL.F + CONTROL.B)) + ((workspace.CurrentCamera.CoordinateFrame * CFrame.new(CONTROL.L + CONTROL.R, (CONTROL.F + CONTROL.B + CONTROL.Q + CONTROL.E) * 0.2, 0).p) - workspace.CurrentCamera.CoordinateFrame.p)) * SPEED
                    lCONTROL = {F = CONTROL.F, B = CONTROL.B, L = CONTROL.L, R = CONTROL.R}
                elseif (CONTROL.L + CONTROL.R) == 0 and (CONTROL.F + CONTROL.B) == 0 and (CONTROL.Q + CONTROL.E) == 0 and SPEED ~= 0 then
                    BV.velocity = ((workspace.CurrentCamera.CoordinateFrame.lookVector * (lCONTROL.F + lCONTROL.B)) + ((workspace.CurrentCamera.CoordinateFrame * CFrame.new(lCONTROL.L + lCONTROL.R, (lCONTROL.F + lCONTROL.B + CONTROL.Q + CONTROL.E) * 0.2, 0).p) - workspace.CurrentCamera.CoordinateFrame.p)) * SPEED
                else
                    BV.velocity = Vector3.new(0, 0, 0)
                end
                BG.cframe = workspace.CurrentCamera.CoordinateFrame
            until not FLYING

            -- Cleanup after flying ends
            CONTROL = {F = 0, B = 0, L = 0, R = 0, Q = 0, E = 0}
            lCONTROL = {F = 0, B = 0, L = 0, R = 0, Q = 0, E = 0}
            SPEED = 0
            BG:Destroy()
            BV:Destroy()
            if Players.LocalPlayer.Character:FindFirstChildOfClass('Humanoid') then
                Players.LocalPlayer.Character:FindFirstChildOfClass('Humanoid').PlatformStand = false
            end
        end)
    end

    -- Start flying
    startFlying()
end

-- Function to handle when a player enters a vehicle
local function onEnterVehicle(vehicle)
    if vehicleFlyEnabled then
        currentVehicle = vehicle
        startVehicleFly(vehicle)
    end
end

-- Function to handle when a player exits a vehicle
local function onExitVehicle()
    if currentVehicle then
        currentVehicle = nil
        FLYING = false  -- Stop any ongoing flying coroutine related to vehicles
    end
end

-- Listen for when a player enters or exits a vehicle
game.Players.LocalPlayer.CharacterAdded:Connect(function(character)
    character.ChildAdded:Connect(function(child)
        if child:IsA("VehicleSeat") then
            child:GetPropertyChangedSignal("Occupant"):Connect(function()
                local vehicle = child.Parent
                local occupant = child.Occupant
                if occupant == game.Players.LocalPlayer then
                    onEnterVehicle(vehicle)
                else
                    onExitVehicle()
                end
            end)
        end
    end)
end)

-- Function to update ESP
local function updateESP()
    -- Clear previous drawings
    for _, drawing in pairs(Drawing.GetAll()) do
        drawing:Remove()
    end
    
    -- Iterate through each player
    for _, player in pairs(game.Players:GetPlayers()) do
        -- Skip if it's the local player
        if player ~= game.Players.LocalPlayer then
            local character = player.Character
            if character and character:FindFirstChild("HumanoidRootPart") then
                local rootPart = character.HumanoidRootPart
                local rootPos = rootPart.Position
                local rootScreenPos, onScreen = workspace.CurrentCamera:WorldToViewportPoint(rootPos)

                if onScreen then
                    -- Calculate dimensions and position of the box
                    local boxSize = Vector3.new(2, 3, 0)  -- Adjust as needed for width, height, and depth
                    local boxPosition = Vector3.new(rootScreenPos.X, rootScreenPos.Y, rootScreenPos.Z)

                    -- Calculate player's health percentage
                    local humanoid = character:FindFirstChildOfClass("Humanoid")
                    local healthPercentage = humanoid.Health / humanoid.MaxHealth

                    -- Determine box color based on health
                    local boxColor = Color3.fromHSV(healthPercentage * 0.35, 1, 1)  -- HSV color for health-based coloration

                    -- Draw the box using Drawing API (3D perspective)
                    local box = Drawing.new("Quad")
                    box.Visible = true
                    box.Thickness = 1
                    box.Color = boxColor
                    box.PointA = Vector2.new(boxPosition.X - boxSize.X / 2, boxPosition.Y - boxSize.Y / 2)
                    box.PointB = Vector2.new(boxPosition.X + boxSize.X / 2, boxPosition.Y - boxSize.Y / 2)
                    box.PointC = Vector2.new(boxPosition.X + boxSize.X / 2, boxPosition.Y + boxSize.Y / 2)
                    box.PointD = Vector2.new(boxPosition.X - boxSize.X / 2, boxPosition.Y + boxSize.Y / 2)

                    -- Update box position and size based on camera view
                    box.Position = rootScreenPos
                    box.Size = Vector2.new(boxSize.X * (workspace.CurrentCamera.ViewportSize.Y / rootScreenPos.Z), boxSize.Y * (workspace.CurrentCamera.ViewportSize.Y / rootScreenPos.Z))
                end
            end
        end
    end
end

-- Function to manage ESP visibility and update
local function toggleESP(bool)
    if bool then
        -- Start updating ESP periodically
        updateESP()
        game:GetService("RunService").RenderStepped:Connect(updateESP)
    else
        -- Stop updating ESP
        game:GetService("RunService").RenderStepped:Disconnect(updateESP)
        -- Clear any existing drawings (if needed)
        for _, drawing in pairs(Drawing.GetAll()) do
            drawing:Remove()
        end
    end
end

local function updateTracers()
    if tracersEnabled then

        if localRootPart then
            for _, player in pairs(players:GetPlayers()) do

                        local line = Drawing.new("Line")

                    local playerScreenPos, onScreen = workspace.CurrentCamera:WorldToViewportPoint(rootPart.Position)
                    local localScreenPos = workspace.CurrentCamera:WorldToViewportPoint(localRootPart.Position)

                    lines[player.Name].From = Vector2.new(localScreenPos.X, localScreenPos.Y)
                    lines[player.Name].To = Vector2.new(playerScreenPos.X, playerScreenPos.Y)

                    if teamColorsEnabled then
                        lines[player.Name].Color = player.TeamColor.Color

                        lines[player.Name].Color = Color3.fromRGB(255, 255, 255) -- default color
                    end

                    lines[player.Name].Visible = true
                    lines[player.Name].Thickness = tracersThickness
                    lines[player.Name].Transparency = onScreen and 1 or 0.5
                end
            end
        end
    else
        for _, line in pairs(lines) do
            line.Visible = false
        end
    end
end
